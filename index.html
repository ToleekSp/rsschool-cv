function sortBit(arr) {
  arr.sort((a, b) => {
    const countBitA = a.toString(2).split('').filter(elm => elm === '1').length;
    const countBitB = b.toString(2).split('').filter(elm => elm === '1').length;

    return countBitA === countBitB ? a - b : countBitA - countBitB;
  });
}
